Pyramid Pattern 2 by C Language

Source Code:

            #include <stdio.h>
            int main()
            {
                int i,j,n;
                scanf("%d", &n);
                for(i=n;i>=1;--i){
                    for(j=n;j>=1;--j){
                        if(j<=i)
                            printf("*");
                        else
                            printf(".");
                    }
                    printf("\n");
                }
                return 0;
            }
