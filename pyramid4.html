            #include <stdio.h>
            int main()
            {
                int i,j,n,cnt =0, sum = 0;
                scanf("%d", &n);
                for(i=n;i>=1;--i){
                    for(j=1;j<=n-i;++j){
                        printf(".");
                    }
                    for(j=1;j<=i;++j){
                        printf("%d",j);
                    }
                    for(j=i-1;j>=1;--j){
                        printf("%d",j);
                    }
                    for(j=1;j<=n-i;++j){
                        printf(".");
                    }
                    printf("\n");
                }
                return 0;
            }
