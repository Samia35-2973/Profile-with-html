Pyramid Pattern 1 by C Language

Source Code:

            #include <stdio.h>>
            int main()
            {
                int i,j,n;
                scanf("%d", &n);
                for(i=1;i<=n;++i){
                    for(j=1;j<=n;++j){
                        if(j<=i)
                            printf("*");
                        else
                            printf(".");
                    }
                    printf("\n");
                }
                return 0;
            }
